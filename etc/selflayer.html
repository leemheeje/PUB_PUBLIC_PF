내장함수
									var appendHtml = function(target) {
										$getUrl.done(function(data) {
											$(target).append(function() {
												$(this).append(data);
												$(this).customTags();
											});
										});
									}
									for (var i = 0; i < include.length; i++) {
										if (include[i]) {
											if (include[i][1].get) {
												var $getUrl = $.get(include[i][1].url);
												var target = include[i][1].target;
												appendHtml(target);
											}
										}
									}
